		// Inicializando la directiva ngWizard
		$scope.wizoptions={
	            show:'all',
	            disabled:true,
	            titleNext:'Siguiente',
	            titleBack:'Anterior'
	        };

	    // Inicializando las directivas ngAlert

	        $scope.alertone={
		  message:'Selecciona una opción para continuar.',
		  type:'info',
		  show:true,
		  hide:{auto:false,click:true,time:0}
		}

		$scope.alerttwo={
		  message:'Ingresa todos los datos y presiona el boton guardar para continuar.',
		  type:'info',
		  show:true,
		  hide:{auto:false,click:true,time:0}
		}

		$scope.alertthree={
		  message:'Realiza una busqueda y presiona el boton guardar para finalizar.',
		  type:'info',
		  show:true,
		  hide:{auto:false,click:true,time:0}
		}

			// Realizando las funciones para los eventos
	        
	        $scope.cambiarOpcion=function(value){

	        	if(value==1){

	        		$scope.wizoptions.disabled=false; // Habilitando el siguiente paso del wizard
	   
	        	}else{

	        		$scope.wizoptions.disabled=true; // Deshabilitando el siguiente paso del wizard

	        	}

	        }

	        $scope.clickOpcion=function(){


	        	$scope.wizoptions.disabled=false; // Habilitando el siguiente paso del wizard
	        	$scope.habilitarClick=true;  // Habilitando el click en las pestañas

	        }

	        $scope.buscarInclude=function(){

	        	// Pasando datos al ngInclude con el attrib "scope="

	        	$scope.nombre="Nombre";
	        	$scope.apellido="Apellido";
	        	$scope.telefono="785563332";

	        }

	        $scope.mostrarAlgunos=function(){
	        	$scope.wizoptions.hide=[1,3]; // Ocultando los pasos 2 y 3 del Wizard
	        }

	        $scope.esconderAlgunos=function(){
	        	$scope.wizoptions.hide=[1,3]; // Mostrando los pasos 2 y 3 del Wizard
	        }

	        $scope.mostrarTodos=function(){
	        	$scope.wizoptions.hide="all"; // Ocultando todos los pasos del Wizard
	        }
